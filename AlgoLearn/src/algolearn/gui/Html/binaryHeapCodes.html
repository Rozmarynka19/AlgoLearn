void up(int kopiec[], int n){
	int x, y, t;
	x = 0;
	y = 1;
	while (y < n){
		if (y + 1 < n && kopiec[y + 1] > kopiec[y]){
			y = y + 1;
		}

		if (kopiec[x] >= kopiec[y]){
			break;
		}

		t = kopiec[x];
		kopiec[x] = kopiec[y];
		kopiec[y] = t;
		x = y;
		y = 2 * x + 1;
	}
}
            
void down(int kopiec[], int n){
	int x, y, t;
	x = n;
	y = (x - 1) / 2;
	while (x > 0 && kopiec[y] < kopiec[x]){
		t = kopiec[x];
		kopiec[x] = kopiec[y];
		kopiec[y] = t;
		x = y;
		y = (x - 1) / 2;
	}
}
            
void dodaj(int kopiec[], int n){
	kopiec[n] = n;
	down(kopiec, n);
}
            
void usun(int kopiec[], int n){
	if (n != 0){
		kopiec[0] = kopiec[n];
		kopiec[n] = -1;
		up(kopiec, n - 1);
	}
}
