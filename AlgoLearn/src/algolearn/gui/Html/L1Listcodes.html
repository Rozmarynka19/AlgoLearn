struct ListaJednokierunkowa {
	wezel* brzeg_HEAD = 0;
    wezel* iterator = brzeg_HEAD;
    
	void wstaw(int klucz, char literka);
	void wyszukaj(int klucz);
	void usun(int klucz);
	void wyswietl() {
		if (iterator == nullptr) {
			cout << "Lista nie istnieje" << endl;
		}
		else {
			while (iterator != nullptr) {
				cout << iterator->kluczWezla << iterator->znak << endl;
				iterator = iterator->nastepnyWezel;
			}
			cout << endl;
		}
		iterator = brzeg_HEAD;
	}
	void wyczysc() {
		while (brzeg_HEAD != nullptr) {
			iterator = brzeg_HEAD;
			brzeg_HEAD = iterator->nastepnyWezel;
			delete iterator;
		}
		iterator = brzeg_HEAD;
	}
};
            
struct wezel {
	int kluczWezla = 0;
	char znak = '0';
	wezel* nastepnyWezel = nullptr;
};
            
void wstaw(int klucz, char literka) {
    while (iterator != nullptr) {
        if (iterator->kluczWezla == klucz) {
            cout << "Element o takim kluczu juz istnieje" << endl;
            return;
        }
        iterator = iterator->nastepnyWezel;
    }
    iterator = brzeg_HEAD;
    wezel* p = new wezel;
    iterator = brzeg_HEAD;
    p->nastepnyWezel = nullptr;
    p->kluczWezla = klucz;
    p->znak = literka;
    if (brzeg_HEAD == nullptr) {
        brzeg_HEAD = p;
        return;
    }
    if (p->kluczWezla < iterator->kluczWezla) {
        p->nastepnyWezel = iterator;
        brzeg_HEAD = p;
        return;
    }
    else {
        while (iterator != nullptr) {
            if (iterator->nastepnyWezel == nullptr) {
                iterator->nastepnyWezel = p;
                break;
            }
            if (iterator->nastepnyWezel->kluczWezla > p->kluczWezla) {
                p->nastepnyWezel = iterator->nastepnyWezel;
                iterator->nastepnyWezel = p;
                break;
            }
            iterator = iterator->nastepnyWezel;
        }
    }
    iterator = brzeg_HEAD;
}
            
void usun(int klucz) {
    wezel* usuwany = nullptr;
    wezel* pomocniczy = brzeg_HEAD;
    while (iterator != nullptr) {
        if (iterator->kluczWezla == klucz) {
            usuwany = iterator;
            break;
        }
        pomocniczy = iterator;
        iterator = iterator->nastepnyWezel;
    }
    if (brzeg_HEAD == usuwany) {
        brzeg_HEAD = usuwany->nastepnyWezel;
    }
    else {
            pomocniczy->nastepnyWezel = usuwany->nastepnyWezel;
            delete usuwany;
    }
    iterator = brzeg_HEAD;
}

    while (iterator != NULL) {

            cout << "Element istnieje" << endl;
            return;
        }
        iterator = iterator->nastepnyWezel;
    }
    cout << "Element nie istnieje lub lista jest pusta" << endl;
    iterator = brzeg_HEAD;
}

    KoloRatunkowe tworzonyElement = new KoloRatunkowe();
    tworzonyElement.topielec = klucz;
    if (brzeg_HEAD == null){
        brzeg_HEAD = tworzonyElement; //rzucamy line do brzegu 
    }
    else {
        KoloRatunkowe iterator = brzeg_HEAD;
        while (iterator.nastepneKoloRatunkowe != null) {
            iterator = iterator.nastepneKoloRatunkowe;
        }
        iterator.nastepneKoloRatunkowe = tworzonyElement;
        tworzonyElement.nastepneKoloRatunkowe = null;
    }
}

    if (numerKolejnosci == 1) {
        KoloRatunkowe wskaznik = brzeg_HEAD;
        brzeg_HEAD = wskaznik.nastepneKoloRatunkowe;
    }
    else if (numerKolejnosci >= 2) {
        int iteratorLiczbowy = 1;
        KoloRatunkowe iteratorPoElementach = brzeg_HEAD;
        while (iteratorPoElementach != null) {
            if ((iteratorLiczbowy + 1) == numerKolejnosci) break;
            iteratorPoElementach = iteratorPoElementach.nastepneKoloRatunkowe;
            iteratorLiczbowy++;
        }
        if (iteratorPoElementach.nastepneKoloRatunkowe.nastepneKoloRatunkowe == null) {
            iteratorPoElementach.nastepneKoloRatunkowe = null;
        }
        else {
            iteratorPoElementach.nastepneKoloRatunkowe = 
                iteratorPoElementach.nastepneKoloRatunkowe.nastepneKoloRatunkowe;
        }
    }
}
